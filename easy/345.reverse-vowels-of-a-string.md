 * [345] 反转字符串中的元音字母
 *

 * Total Submissions: 13.7K
 * Testcase Example:  '"hello"'
 *
 * 编写一个函数，以字符串作为输入，反转该字符串中的元音字母。
 * 
 * 示例 1:
 * 
 * 输入: "hello"
 * 输出: "holle"
 * 
 * 
 * 示例 2:
 * 
 * 输入: "leetcode"
 * 输出: "leotcede"
 * 
 * 说明:
 * 元音字母不包含字母"y"。
 * 
 */
class Solution {
    public String reverseVowels(String s) {
        char[] c = s.toCharArray();

        int i = 0;
        int j = c.length - 1;
        while (i < j) {
            char temp = c[0];

            if (!isY(c[i])) {
                i++;
            } else if (!isY(c[j])) {
                j--;
            } else {
                temp = c[i];
                c[i] = c[j];
                c[j] = temp;
                i++;
                j--;
            }
        }

        return new String(c);
    }

    public boolean isY(char c) {
        if (c == 'a'
                || c == 'e'
                || c == 'i'
                || c == 'o'
                || c == 'u'
                || c == 'A'
                || c == 'E'
                || c == 'I'
                || c == 'O'
                || c == 'U'
        ) {
            return true;
        } else {
            return false;
        }
    }
}
